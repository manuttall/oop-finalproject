scene

scene package: Core classes for scene management and 3D camera rendering.

This package provides the main components needed to build, manage, and render 3D scenes, including camera control, screen output, file imports, and scene organization.

The following classes are re-exported for convenient access:

  • AspectRatio: Represents the screen's aspect ratio
  • Camera: Defines the viewpoint and projection system in 3D space
  • Scene: Holds a collection of Mesh3D objects and the active Camera
  • Screen: Manages the Tkinter window and draws the 2D projections

Example: from scene import Camera, Scene, Screen

 18    def __init__(self, origin: Vertex, look_at: Vertex) -> None:
 19        """Constructor
 20
 21        Args:
 22            origin (Vertex): Camera's position in 3D space.
 23            look_at (Vertex): Point the camera looks at.
 24        """
 25        super().__init__(origin.x, origin.y, origin.z)
 26        self._look_at: Vertex = look_at
 27        self._forward: Vector = Vector(0.0, 0.0, -1.0)
 28        self._up: Vector = Vector(0.0, 1.0, 0.0)
 29        self._right: Vector = Vector(1.0, 0.0, 0.0)
 30
 31        self._recalculate_axes()
 32
 33    def _recalculate_axes(self) -> None:
 34        """Recalculates the camera's forward, right, and up vectors
 35        based on current look_at."""
 36        temp_vec = Vector(self._look_at.x, self._look_at.y, self._look_at.z)
 37        forward: Vector = (temp_vec - self).normalize
 38        world_up = Vector(0.0, 0.0, 1.0)
 39
 40        if abs(forward.dot(world_up)) >= 0.999:
 41            world_up = Vector(0.0, 1.0, 0.0)
 42
 43        right: Vector = forward.cross(world_up).normalize
 44        up: Vector = right.cross(forward).normalize
 45
 46        self._forward = forward
 47        self._right = right
 48        self._up = up
 49
 50    def set_look_at(self, new_look_at: Vertex) -> None:
 51        """Sets a new look-at point for the camera and recalculates axes.
 52
 53        Args:
 54            new_look_at (Vertex): New target point for the camera.
 55        """
 56        self._look_at = new_look_at
 57        self._recalculate_axes()

 86    def is_vertex_in_front(self, vertex: Vertex) -> bool:

109    def project_vertex(self, vertex: Vertex) -> Point:
110        """Projects a 3D vertex onto the camera's 2D space.
111
112        Args:
113            vertex (Vertex): 3D vertex to project.
114
115        Returns:
116            Point: 2D projected point.
117        """
118        to_vertex: Vector = vertex - self
119        x_cam: float = to_vertex.dot(self._right)
120        y_cam: float = to_vertex.dot(self._up)
121        z_cam: float = to_vertex.dot(self._forward)
122
123        x_proj = x_cam / z_cam
124        y_proj = y_cam / z_cam
125
126        return Point(x_proj, y_proj)
127
128    def project_face(self, face: Face3D) -> Face2D:
129        """Projects a 3D face onto 2D space.
130
131        Args:
132            face (Face3D): 3D face.
133
134        Returns:
135            Face2D: 2D projection of the face.
136        """
137        projected_points: List[Point] = [
138            self.project_vertex(vertex) for vertex in face.points
139        ]
140        dist: float = face.distance(self)
141        color: Shader = face.color
142        return Face2D(projected_points, dist, color)

 19    """A screen to render 2D triangles (Face2D) on a Tkinter canvas."""

 21    def __init__(self, aspect_ratio: AspectRatio, resolution: int,
 22                 background: Shader = Shader(0, 0, 0)) -> None:
 23        """Constructor
 24
 25        Args:
 26            aspect_ratio (AspectRatio): The aspect ratio of the canvas.
 27            resolution (int): Scaling factor for canvas size.
 28            background (Shader, optional): Background Shader color. Defaults to black.
 29        """
 30        self._aspect_ratio: AspectRatio = aspect_ratio
 31        self._resolution: int = resolution
 32        self._canvas_width: int = int(aspect_ratio.horizontal * resolution)
 33        self._canvas_height: int = int(aspect_ratio.vertical * resolution)
 34        self._background: str = background.hex
 35
 36        self._window: Optional[tk.Tk] = None
 37        self._canvas: Optional[tk.Canvas] = None
 38
 39    def _create_canvas(self) -> None:
 40        """Creates the Tkinter window and canvas."""
 41        self._window = tk.Tk()
 42        self._window.title("3D Renderer")
 43        self._canvas = tk.Canvas(
 44            self._window,
 45            width=self._canvas_width,
 46            height=self._canvas_height,
 47            bg=self._background
 48        )
 49        self._canvas.pack()
 50
 51    def _translate_point(self, point: Point) -> Point:
 52        """Converts a normalized projected Point to canvas-space Point.
 53
 54        Args:
 55            point (Point): Projected point in camera space.
 56
 57        Returns:
 58            Point: Pixel coordinates for the canvas.
 59        """
 60        x_canvas = int((point.x + self._aspect_ratio.horizontal / 2) * self._resolution)
 61        y_canvas = int((-point.y + self._aspect_ratio.vertical / 2) * self._resolution)
 62        return Point(x_canvas, y_canvas)
 63
 64    def _draw_face(self, face: Face2D) -> None:
 65        """Draws a single Face2D triangle on the canvas.
 66
 67        Args:
 68            face (Face2D): The face to draw.
 69
 70        Raises:
 71            RuntimeError: If the canvas has not been created yet.
 72        """
 73        if self._canvas is None:
 74            raise RuntimeError("Canvas not initialized.")
 75
 76        translated_points = [self._translate_point(p) for p in face.points]
 77        coords = [(p.x, p.y) for p in translated_points]
 78
 79        self._canvas.create_polygon(
 80            coords,
 81            fill=face.color.hex,
 82            outline=""
 83        )
 84
 85    def _draw_faces(self, faces: List[Face2D]) -> None:
 86        """Draws a sorted list of visible Face2D triangles from farthest to nearest.
 87
 88        Args:
 89            faces (List[Face2D]): The 2D faces to draw.
 90        """
 91        sorted_faces = sorted(faces, reverse=True)  # Draw farthest faces first
 92        for face2d in sorted_faces:
 93            self._draw_face(face2d)
 94
 95    def show(self) -> None:
 96        """Displays the window and starts the main event loop."""
 97        if self._window is None:
 98            self._create_canvas()
 99        if self._window is not None:
100            self._window.mainloop()
101
102    def render(self, faces: List[Face2D]) -> None:
103        """Renders a list of Face2D objects by creating the canvas, drawing them,
104        and showing the window.
105
106        Args:
107            faces (List[Face2D]): The 2D faces to render.
108        """
109        self._create_canvas()
110        self._draw_faces(faces)
